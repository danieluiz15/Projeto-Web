CREATE DATABASE IF NOT EXISTS pet;

USE pet;

CREATE TABLE pet_adocao(
	id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(60) NOT NULL,
    tipo ENUM("Cachorro", "Gato", "Outro") NOT NULL,
    idade DECIMAL(4,2) NOT NULL,
    descricao VARCHAR(255) NOT NULL
);

CREATE TABLE adocao(
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(60) NOT NULL,
    sobrenome VARCHAR(60) NOT NULL,
    endereco VARCHAR(100) NOT NULL,
    endereco_aux VARCHAR(100) NOT NULL,
    cidade VARCHAR(30) NOT NULL,
    estado ENUM("MJ", "DF", "SP", "RJ") NOT NULL,
    cep VARCHAR(9) NOT NULL,
    petID INT NOT NULL,
    motivo VARCHAR(255) NOT NULL,
    FOREIGN KEY (petID) REFERENCES pet_adocao(id)
);
